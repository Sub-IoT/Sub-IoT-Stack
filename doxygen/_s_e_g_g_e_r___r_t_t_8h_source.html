82 //
83 // Description for a circular buffer (also called "ring buffer")
84 // which is used as up-buffer (T->H)
85 //
86 typedef struct {
87  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
88  char* pBuffer; // Pointer to start of buffer
89  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
90  unsigned WrOff; // Position of next item to be written by either target.
91  volatile unsigned RdOff; // Position of next item to be read by host. Must be volatile since it may be modified by host.
92  unsigned Flags; // Contains configuration flags
93 } SEGGER_RTT_BUFFER_UP;
94 
95 //
96 // Description for a circular buffer (also called "ring buffer")
97 // which is used as down-buffer (H->T)
98 //
99 typedef struct {
100  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
101  char* pBuffer; // Pointer to start of buffer
102  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
103  volatile unsigned WrOff; // Position of next item to be written by host. Must be volatile since it may be modified by host.
104  unsigned RdOff; // Position of next item to be read by target (down-buffer).
105  unsigned Flags; // Contains configuration flags
106 } SEGGER_RTT_BUFFER_DOWN;
107 
108 //
109 // RTT control block which describes the number of buffers available
110 // as well as the configuration for each buffer
111 //
112 //
113 typedef struct {
114  char acID[16]; // Initialized to "SEGGER RTT"
115  int MaxNumUpBuffers; // Initialized to SEGGER_RTT_MAX_NUM_UP_BUFFERS (type. 2)
116  int MaxNumDownBuffers; // Initialized to SEGGER_RTT_MAX_NUM_DOWN_BUFFERS (type. 2)
117  SEGGER_RTT_BUFFER_UP aUp[SEGGER_RTT_MAX_NUM_UP_BUFFERS]; // Up buffers, transferring information up from target via debug probe to host
118  SEGGER_RTT_BUFFER_DOWN aDown[SEGGER_RTT_MAX_NUM_DOWN_BUFFERS]; // Down buffers, transferring information down from host via debug probe to target
119 } SEGGER_RTT_CB;
