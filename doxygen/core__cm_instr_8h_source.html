503 __attribute__( ( always_inline ) ) __STATIC_INLINE uint8_t __LDREXB(volatile uint8_t *addr)
504 {
505  uint32_t result;
506 
507 #if (__GNUC__ > 4) || (__GNUC__ == 4 && __GNUC_MINOR__ >= 8)
508  __ASM volatile ("ldrexb %0, %1" : "=r" (result) : "Q" (*addr) );
509 #else

514 #endif
