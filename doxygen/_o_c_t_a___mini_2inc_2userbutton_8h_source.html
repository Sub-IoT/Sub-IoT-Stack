21  * Interface to the hardware buttons of the gecko board. This file is NOT a part of the
22  * 'HAL' interface since not every platform will have buttons
23  *
24  * The EFM32GG 'Giant Geck' board has two userbuttons. PB0 and PB1.
25  * These buttons are identified using a button_id_t.
26  * - PB0 is identified with button id '0'
27  * - PB1 is identified with button id '1'
28  *
29  */

47 /* \brief Check whether a button is currently pressed or not.
48  *
49  * if an invalid button_id is supplied, this function returns false
50  *
51  * \return bool TRUE if the button is pressed
52  * FALSE if the button is not pressed or if an invalid button id was supplied
53  * */

56 /* \brief Register a function to be called when the button with the specified <button_id> is pressed
57  *
58  * Multiple callback functions can be registered for the same button but the same function cannot be registered twice.
59  * If a previously registered callback is re-registered for the same button, EALREADY is returned
60  *
61  * \param button_id The id of the button for which to register the callback '0' for PB0, '1' for PB1
62  * \param callback The function to call when the button is pressed
63  * \return error_t SUCCESS if the callback was successfully registered
64  * ESIZE if an invalid button_id was specified
65  * EINVAL if callback is 0x0
66  * EALREADY if the callback was already registered for this button
67  * ENOMEM if the callback could not be registered because there are already too many
68  * callbacks registered for this button
69  */
70 __LINK_C error_t ubutton_register_callback(button_id_t button_id, ubutton_callback_t callback);
71 
72 /* \brief Deregister a callback function previously registered using 'register_button_callback'
73  *
74  * \param button_id The id of the button for which to register the callback '0' for PB0, '1' for PB1
75  * \param callback The function to deregister
76  * \return error_t SUCCESS if the callback was successfully deregistered
77  * ESIZE if an invalid button_id was specified
78  * EINVAL if callback is 0x0
79  * EALREADY if the callback was not registered for this button
80  */
81 __LINK_C error_t ubutton_deregister_callback(button_id_t button_id, ubutton_callback_t callback);
