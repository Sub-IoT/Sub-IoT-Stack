IF(HAL_RADIO_USE_HW_CRC)
    MESSAGE("netdev does not support (DASH7 compatible) hardware CRC, forcing HAL_RADIO_USE_HW_CRC to FALSE")
    SET(HAL_RADIO_USE_HW_CRC "FALSE" CACHE BOOL "Enable/Disable the use of HW CRC" FORCE)
ENDIF()

INCLUDE_DIRECTORIES(
    inc
)

#Export the 'inc' directory globally		    
EXPORT_GLOBAL_INCLUDE_DIRECTORIES(
    inc
)

SET(netdev_driver_SRC
    netdev.c
    sx127x.c
    sx127x_internal.c
    sx127x_netdev.c
    sx127x_getset.c
)


#An object library with name '${CHIP_LIBRARY_NAME}' MUST be generated by the CMakeLists.txt file for every chip
ADD_LIBRARY (${CHIP_LIBRARY_NAME} OBJECT ${netdev_driver_SRC})
